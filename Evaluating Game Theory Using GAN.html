# Payoff Matrix for Prisoner's Dilemma
payoff_matrix = {
    (0, 0): (3, 3),  # Cooperate, Cooperate
    (0, 1): (0, 5),  # Cooperate, Defect
    (1, 0): (5, 0),  # Defect, Cooperate
    (1, 1): (1, 1)   # Defect, Defect
}

@lru_cache(maxsize=None)  # Memoize the payoff function
def get_payoff(action1, action2):
    return payoff_matrix[(action1, action2)]

# Modified game simulation with corrected strategy implementations
def play_game(generator, strategy, rounds=50):
    total_payoff_gen, total_payoff_strat = 0, 0
    gen_moves = []  # Initialize as empty list
    strat_moves = []  # Initialize as empty list
    history_gen = []
    history_strat = []

    # Strategy-specific state trackers
    grim_triggered = False
    wsls_prev_action = 0  # Win-Stay Lose-Shift initial action

    for _ in range(rounds):
        # Get state from history
        last_move_gen = history_gen[-1] if len(history_gen) > 0 else 0
        last_move_strat = history_strat[-1] if len(history_strat) > 0 else 0

        # Generator's move
        gen_input = np.array([[last_move_gen, last_move_strat]])
        gen_move = 1 if generator.predict(gen_input, verbose=0)[0][0] > 0.5 else 0

        # Opponent's move with corrected implementations
        if strategy == "Always Cooperate":
            strat_move = 0
        elif strategy == "Always Defect":
            strat_move = 1
        elif strategy == "Tit-for-Tat":
            strat_move = last_move_gen
        elif strategy == "Grim Trigger":
            if grim_triggered:
                strat_move = 1
            else:
                strat_move = 0
                if gen_move == 1:
                    grim_triggered = True
        elif strategy == "Random":
            strat_move = random.choice([0, 1])
        elif strategy == "Win-Stay Lose-Shift":
            if len(history_strat) > 0:
                # Use opponent's previous payoff to decide
                last_payoff = get_payoff(history_gen[-1], history_strat[-1])[1]
                if last_payoff >= 3:  # Win-Stay
                    strat_move = history_strat[-1]
                else:  # Lose-Shift
                    strat_move = 1 - history_strat[-1]
            else:
                strat_move = 0  # Initial cooperation

        # Record moves and update payoffs
        payoff_gen, payoff_strat = get_payoff(gen_move, strat_move)
        total_payoff_gen += payoff_gen
        total_payoff_strat += payoff_strat

        gen_moves.append(gen_move)
        strat_moves.append(strat_move)
        history_gen.append(gen_move)
        history_strat.append(strat_move)

    return total_payoff_gen, total_payoff_strat, gen_moves, strat_moves

# Improved GAN training with proper adversarial setup
def train_gan(generator, discriminator, strategies, epochs=100, batch_size=32):
    gen_losses, disc_losses = [], []
    expert_strategies = ["Tit-for-Tat", "Grim Trigger", "Win-Stay Lose-Shift"]  # Optimal strategies

    # Define the training step as a tf.function for potential speedup
    @tf.function
    def train_step(real_states, real_actions):
        # Generate fake actions using current generator
        fake_actions = generator(real_states)

        # Prepare discriminator data
        real_data = tf.concat([real_states, tf.cast(tf.reshape(real_actions, (-1, 1)), tf.float32)], axis=1)
        fake_data = tf.concat([real_states, fake_actions], axis=1)

        # Train discriminator
        with tf.GradientTape() as disc_tape:
            prediction_real = discriminator(real_data)
            prediction_fake = discriminator(fake_data)
            d_loss_real = tf.keras.losses.binary_crossentropy(tf.ones_like(prediction_real), prediction_real)
            d_loss_fake = tf.keras.losses.binary_crossentropy(tf.zeros_like(prediction_fake), prediction_fake)
            d_loss = 0.5 * (tf.reduce_mean(d_loss_real) + tf.reduce_mean(d_loss_fake))
        disc_gradients = disc_tape.gradient(d_loss, discriminator.trainable_variables)
        discriminator.optimizer.apply_gradients(zip(disc_gradients, discriminator.trainable_variables))

        # Train generator
        with tf.GradientTape() as gen_tape:
            generated_actions = generator(real_states)
            combined_data = tf.concat([real_states, generated_actions], axis=1)
            predictions = discriminator(combined_data)
            g_loss = tf.reduce_mean(tf.keras.losses.binary_crossentropy(tf.ones_like(predictions), predictions))
        gen_gradients = gen_tape.gradient(g_loss, generator.trainable_variables)

        # Apply gradient clipping
        gen_gradients = [tf.clip_by_norm(g, 1.0) for g in gen_gradients]
        discriminator.optimizer.apply_gradients(zip(disc_gradients, discriminator.trainable_variables))

        generator.optimizer.apply_gradients(zip(gen_gradients, generator.trainable_variables))

        return g_loss, d_loss

    for epoch in range(epochs):
        # Collect real data from expert strategies
        real_states = []
        real_actions = []

        # Generate expert gameplay data
        for _ in range(batch_size):
            expert = random.choice(expert_strategies)
            _, _, _, expert_moves = play_game(generator, expert, rounds=30)
            history = []
            for i in range(len(expert_moves)):
                if i == 0:
                    # Initial state (no history)
                    state = [0, 0]  # Default initial state
                else:
                    # Use previous moves as state
                    state = [expert_moves[i-1], history[i-1][0]]

                action = expert_moves[i]
                real_states.append(state)
                real_actions.append(action)
                history.append(state)  # Update history

        real_states = np.array(real_states[:batch_size], dtype=np.float32)
        real_actions = np.array(real_actions[:batch_size], dtype=np.int32) # Use int32 for actions

        # Train the GAN using the tf.function
        g_loss, d_loss = train_step(real_states, real_actions)

        # Record losses
        gen_losses.append(g_loss.numpy())
        disc_losses.append(d_loss.numpy())

    return gen_losses, disc_losses
